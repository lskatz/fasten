//! A simple statistics library
//!
//! Heavily inspired by the python standard library statistics module.

extern crate rand;
extern crate num;

mod univariate_;
mod stats_;

pub mod univariate {
    pub use univariate_::{
        harmonic_mean,
        geometric_mean,
        quadratic_mean,
        mode,
        average_deviation,
        pearson_skewness,
        skewness,
        pskewness,
        kurtosis,
        pkurtosis,
        standard_error_mean,
        standard_error_skewness,
        standard_error_kurtosis
    };
}

pub use univariate::mode;
pub use stats_::{
    Degree,
    mean,
    median,
    variance,
    population_variance,
    standard_deviation,
    population_standard_deviation,
    standard_scores
};
